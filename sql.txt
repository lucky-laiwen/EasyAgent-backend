-- 下属为此项目所需要的全部数据库，粘贴到数据库管理工具(Navicat DataGrip)运行即可
-- 创建数据库（如果尚未创建）
CREATE DATABASE IF NOT EXISTS agent;
USE agent;

-- 1. user 表（无外键依赖）
CREATE TABLE user (
    id INT PRIMARY KEY AUTO_INCREMENT,
    name VARCHAR(255) UNIQUE NOT NULL,
    email VARCHAR(255) NOT NULL,
    password VARCHAR(255) NOT NULL,
    is_active INT DEFAULT 0,
    avatar TEXT DEFAULT 'https://images.unsplash.com/photo-1506744038136-46273834b3fb?auto=format&fit=crop&w=1200&q=80',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_id (id)
);

-- 2. chat 表（依赖 user）
CREATE TABLE chat (
    id INT PRIMARY KEY AUTO_INCREMENT,
    title TEXT,
    user_id INT NOT NULL,
    kb_id INT DEFAULT NULL COMMENT '绑定的知识库ID',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_chat_id (id),
    FOREIGN KEY (user_id) REFERENCES user(id) ON DELETE CASCADE
);

-- 3. messages 表（依赖 chat）
CREATE TABLE messages (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    chat_id INT NOT NULL,
    sender TINYINT NOT NULL COMMENT '0=user, 1=ai',
    content TEXT NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    think_content TEXT,
    message_type VARCHAR(10) DEFAULT NULL COMMENT 'text/ppt, NULL=text',
    rag_references JSON DEFAULT NULL COMMENT 'RAG引用的文档信息',
    INDEX idx_messages_id (id),
    FOREIGN KEY (chat_id) REFERENCES chat(id) ON DELETE CASCADE
);

-- 3.1 tool_calls 表（依赖 messages）
CREATE TABLE tool_calls (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    message_id BIGINT NOT NULL,
    tool_name VARCHAR(100) NOT NULL COMMENT '工具名称',
    tool_content MEDIUMTEXT COMMENT '工具返回结果',
    tool_input TEXT COMMENT '工具输入参数',
    status INT DEFAULT 1 COMMENT '状态: 1=成功, 0=失败, 2=进行中',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_tool_calls_id (id),
    FOREIGN KEY (message_id) REFERENCES messages(id) ON DELETE CASCADE
);

-- 4. chat_message 表（依赖 user）
CREATE TABLE chat_message (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    sender_id INT NOT NULL,
    receiver_id INT NOT NULL,
    content TEXT NOT NULL,
    status TINYINT DEFAULT 0 COMMENT '0=未读,1=已读',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    share_chat_id INT,
    INDEX idx_chat_message_id (id),
    FOREIGN KEY (sender_id) REFERENCES user(id) ON DELETE CASCADE,
    FOREIGN KEY (receiver_id) REFERENCES user(id) ON DELETE CASCADE
);

-- 5. chat_share 表（依赖 chat 和 user）
CREATE TABLE chat_share (
    id INT PRIMARY KEY AUTO_INCREMENT,
    chat_id INT NOT NULL,
    owner_id INT NOT NULL,
    shared_to_id INT NOT NULL,
    permission INT DEFAULT 1,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_chat_share_id (id),
    FOREIGN KEY (chat_id) REFERENCES chat(id) ON DELETE CASCADE,
    FOREIGN KEY (owner_id) REFERENCES user(id) ON DELETE CASCADE,
    FOREIGN KEY (shared_to_id) REFERENCES user(id) ON DELETE CASCADE
);

-- 6. user_friend 表（依赖 user）
CREATE TABLE user_friend (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    friend_id INT NOT NULL,
    status TINYINT DEFAULT 0 COMMENT '0=待确认,1=已确认,2=被拉黑',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_user_friend_id (id),
    FOREIGN KEY (user_id) REFERENCES user(id) ON DELETE CASCADE,
    FOREIGN KEY (friend_id) REFERENCES user(id) ON DELETE CASCADE
);

-- 7. system_message 表（依赖 user）
CREATE TABLE system_message (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    source_id INT,
    title VARCHAR(255),
    content TEXT,
    is_read INT DEFAULT 0,
    action_type INT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_system_message_id (id),
    FOREIGN KEY (user_id) REFERENCES user(id) ON DELETE CASCADE,
    FOREIGN KEY (source_id) REFERENCES user(id) ON DELETE CASCADE
);

-- 8. knowledge_base 表（依赖 user）
CREATE TABLE knowledge_base (
    id INT PRIMARY KEY AUTO_INCREMENT,
    user_id INT NOT NULL,
    name VARCHAR(100) NOT NULL,
    description VARCHAR(500) DEFAULT '',
    chroma_collection VARCHAR(100) NOT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_knowledge_base_id (id),
    FOREIGN KEY (user_id) REFERENCES user(id) ON DELETE CASCADE
);

-- 9. knowledge_document 表（依赖 knowledge_base、chat、messages）
CREATE TABLE knowledge_document (
    id INT PRIMARY KEY AUTO_INCREMENT,
    kb_id INT NOT NULL,
    chat_id INT DEFAULT NULL COMMENT 'NULL=全局文档, 非NULL=会话专属文档',
    message_id BIGINT DEFAULT NULL COMMENT 'NULL=未绑定消息, 非NULL=绑定到具体消息',
    filename VARCHAR(255) NOT NULL,
    file_type VARCHAR(20) NOT NULL,
    chunk_count INT DEFAULT 0,
    status VARCHAR(20) DEFAULT 'processing' COMMENT 'processing/completed/failed',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    updated_at DATETIME DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
    INDEX idx_knowledge_document_id (id),
    FOREIGN KEY (kb_id) REFERENCES knowledge_base(id) ON DELETE CASCADE,
    FOREIGN KEY (chat_id) REFERENCES chat(id) ON DELETE CASCADE,
    FOREIGN KEY (message_id) REFERENCES messages(id) ON DELETE CASCADE
);

-- 10. chat_attachment 表（依赖 chat、messages）
CREATE TABLE chat_attachment (
    id BIGINT PRIMARY KEY AUTO_INCREMENT,
    chat_id INT DEFAULT NULL,
    message_id BIGINT DEFAULT NULL,
    filename VARCHAR(255) NOT NULL,
    file_type VARCHAR(20) NOT NULL,
    file_size BIGINT NOT NULL,
    minio_key VARCHAR(500) NOT NULL,
    file_url VARCHAR(1000) DEFAULT NULL,
    text_content TEXT DEFAULT NULL,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    INDEX idx_chat_attachment_id (id),
    FOREIGN KEY (chat_id) REFERENCES chat(id) ON DELETE CASCADE,
    FOREIGN KEY (message_id) REFERENCES messages(id) ON DELETE SET NULL
);
